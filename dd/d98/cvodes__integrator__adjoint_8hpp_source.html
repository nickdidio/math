406  void chain() final {
407  if (!is_var_return_) {
408  return;
409  }
410 
411  // for sensitivities wrt to ts we do not need to run the backward
412  // integration
413  if (is_var_ts_) {
414  Eigen::VectorXd step_sens = Eigen::VectorXd::Zero(N_);
415  for (int i = 0; i < ts_.size(); ++i) {
416  for (int j = 0; j < N_; ++j) {
417  step_sens.coeffRef(j)
418  += forward_as<var>(solver_->y_return_[i].coeff(j)).adj();
419  }
420 
421  adjoint_of(ts_[i]) += step_sens.dot(
422  rhs(value_of(ts_[i]), y_[i], solver_->value_of_args_tuple_));
423  step_sens.setZero();
424  }
425 
426  if (is_var_only_ts_) {
427  return;
428  }
429  }
430 
431  state_backward_.setZero();
432  quad_.setZero();
433 
434  // At every time step, collect the adjoints from the output
435  // variables and re-initialize the solver
436  double t_init = value_of(ts_.back());
437  for (int i = ts_.size() - 1; i >= 0; --i) {
438  // Take in the adjoints from all the output variables at this point
439  // in time
440  for (int j = 0; j < N_; ++j) {
441  state_backward_.coeffRef(j)
442  += forward_as<var>(solver_->y_return_[i].coeff(j)).adj();
443  }
444 
445  double t_final = value_of((i > 0) ? ts_[i - 1] : t0_);
446  if (t_final != t_init) {
447  if (unlikely(!backward_is_initialized_)) {
448  check_flag_sundials(CVodeCreateB(solver_->cvodes_mem_,
449  solver_backward_, &index_backward_),
450  "CVodeCreateB");

454  reinterpret_cast<void*>(this)),
455  "CVodeSetUserDataB");
456

458  // the states of the backward problem *are* the adjoints
459  // of the ode states

560  }
561

563  if (is_any_var_args_) {
564  for (size_t s = 0; s < num_args_vars_; ++s) {
565  args_varis_[s]->adj_ += quad_.coeff(s);
566  }
567  }

613  /*
614  * Calculate the adjoint sensitivity RHS for varying initial conditions
615  * and parameters
616  *
617  * Equation 2.23 in the cvs_guide.
618  *
619  * @param[in] t time
620  * @param[in] y state of the base ODE system
621  * @param[in] yB state of the adjoint ODE system
622  * @param[out] yBdot evaluation of adjoint ODE RHS
623  */
624  inline int rhs_adj(double t, N_Vector y, N_Vector yB, N_Vector yBdot) const {
625  const nested_rev_autodiff nested;
626 
627  Eigen::Matrix<var, Eigen::Dynamic, 1> y_vars(
628  Eigen::Map<const Eigen::VectorXd>(NV_DATA_S(y), N_));
629  Eigen::Matrix<var, Eigen::Dynamic, 1> f_y_t_vars
630  = rhs(t, y_vars, solver_->value_of_args_tuple_);
631  check_size_match(solver_->function_name_str_.c_str(), "dy_dt",
632  f_y_t_vars.size(), "states", N_);
633  f_y_t_vars.adj() = -Eigen::Map<Eigen::VectorXd>(NV_DATA_S(yB), N_);
634  grad();
635  Eigen::Map<Eigen::VectorXd>(NV_DATA_S(yBdot), N_) = y_vars.adj();
636  return 0;

649  * Calculate the RHS for the quadrature part of the adjoint ODE
650  * problem.
651  *
652  * This is the integrand of equation 2.22 in the cvs_guide.
653  *
654  * @param[in] t time
655  * @param[in] y state of the base ODE system
656  * @param[in] yB state of the adjoint ODE system
657  * @param[out] qBdot evaluation of adjoint ODE quadrature RHS
658  */
659  inline int quad_rhs_adj(double t, N_Vector y, N_Vector yB, N_Vector qBdot) {
660  Eigen::Map<const Eigen::VectorXd> y_vec(NV_DATA_S(y), N_);
661  const nested_rev_autodiff nested;
662 
663  // The vars here do not live on the nested stack so must be zero'd
664  // separately
665  stan::math::for_each([](auto&& arg) { zero_adjoints(arg); },
666  solver_->local_args_tuple_);
667  Eigen::Matrix<var, Eigen::Dynamic, 1> f_y_t_vars
668  = rhs(t, y_vec, solver_->local_args_tuple_);
669  check_size_match(solver_->function_name_str_.c_str(), "dy_dt",
670  f_y_t_vars.size(), "states", N_);
671  f_y_t_vars.adj() = -Eigen::Map<Eigen::VectorXd>(NV_DATA_S(yB), N_);
672  grad();
673  apply(
674  [&qBdot](auto&&... args) {
675  accumulate_adjoints(NV_DATA_S(qBdot), args...);
676  },
677  solver_->local_args_tuple_);
678  return 0;
679  }

695  Eigen::Map<Eigen::MatrixXd> Jfy(SM_DATA_D(J), N_, N_);
696 
697  nested_rev_autodiff nested;
698 
699  Eigen::Matrix<var, Eigen::Dynamic, 1> y_var(
700  Eigen::Map<const Eigen::VectorXd>(NV_DATA_S(y), N_));
701  Eigen::Matrix<var, Eigen::Dynamic, 1> fy_var
702  = rhs(t, y_var, solver_->value_of_args_tuple_);
703 
704  check_size_match(solver_->function_name_str_.c_str(), "dy_dt",
705  fy_var.size(), "states", N_);
706 
707  grad(fy_var.coeffRef(0).vi_);
708  Jfy.col(0) = y_var.adj();
709  for (int i = 1; i < fy_var.size(); ++i) {
710  nested.set_zero_all_adjoints();
711  grad(fy_var.coeffRef(i).vi_);
712  Jfy.col(i) = y_var.adj();
713  }
714  Jfy.transposeInPlace();
715  return 0;
